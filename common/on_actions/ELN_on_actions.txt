on_actions = {
	on_startup = {
		effect = {
			ELN = {
				country_event = {
					id = elino.1
					days = 10
				}
				if = {
					limit = {
						not = {
							has_idea = ELN_Master_of_eu
						}
					}
					add_ideas = ELN_Master_of_eu
				}
			}
		}
	}
	on_offer_join_faction = {
		effect = {
			if = {
				limit = {
					tag = ELN
				}
				save_global_event_target_as = ELN_faction_leader
				FROM = {
					save_global_event_target_as = ELN_faction_member
				}
				set_global_flag = ELN_faction_joining_window_visible
				sound_effect = "peace_summary_message"
			}
		}
	}
	on_monthly_ELN = {
		effect = {
			if = {
				limit = {
					has_global_flag = Entente_for_Democracies_faction
				}
				Entente_for_Democracies_array_effect = yes
			}
		}
	}
}
